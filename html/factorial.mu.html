10 def factorial n:num -> result:num [
11   local-scope
12   load-inputs
13   {
14     # if n=0 return 1
15     zero?:bool <- equal n, 0
16     break-unless zero?
17     return 1
18   }
19   # return n * factorial(n-1)
20   x:num <- subtract n, 1
21   subresult:num <- factorial x
22   result <- multiply subresult, n
23 ]
