 87 void test_split_works() {
 88   vector<string> result = split("abc,def", ",");
 89   CHECK_EQ(result.size(), 2);
 90   CHECK_EQ(result.at(0), "abc");
 91   CHECK_EQ(result.at(1), "def");
 92 }
